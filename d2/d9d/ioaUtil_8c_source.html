104  *pDot = 0;

106  // find the last occurrance of slash
107  for ( pDot = Buffer + strlen(Buffer) - 1; pDot >= Buffer; pDot-- )
108  if (!((*pDot >= '0' && *pDot <= '9') ||
109  (*pDot >= 'a' && *pDot <= 'z') ||
110  (*pDot >= 'A' && *pDot <= 'Z') ||
111  *pDot == '_' || *pDot == '.') )
112  break;
113  return pDot + 1;

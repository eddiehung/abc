83 // blockSize = ShiftSize/4
84 int firstShiftWithOneBit(word x, int blockSize)
85 {
86  int n = 0;
87  if(blockSize == 16){ return 0;}
88  if (x >= ABC_CONST(0x0000000100000000)) {n = n + 32; x = x >> 32;}

91  if(blockSize == 4){ return (64-n)/16;}

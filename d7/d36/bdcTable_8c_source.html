77  if ( Bdc_TableCheckContainment( p, pIsf, pFunc->puFunc ) )
78  return pFunc;
79  Bdc_IsfNot( pIsf );

81  if ( Bdc_TableCheckContainment( p, pIsf, pFunc->puFunc ) )
82  {
83  Bdc_IsfNot( pIsf );
84  return Bdc_Not(pFunc);
85  }
86  Bdc_IsfNot( pIsf );

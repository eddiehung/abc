33  for(var = 0; var < cube.num_binary_vars; var++)
34  cost->in += cdata.var_zeros[var];
35

40  else

42

44  if (cube.num_binary_vars != cube.num_vars) {
45  var = cube.num_vars - 1;
46  cost->out = F->count * cube.part_size[var] - cdata.var_zeros[var];
47  }
48

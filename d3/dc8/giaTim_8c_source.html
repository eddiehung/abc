472  Tim_Man_t * pManTime = (Tim_Man_t *)p->pManTime;
473  Gia_Obj_t * pObj, * pObjIn;

494  int nBoxOutputs = Tim_ManBoxOutputNum( pManTime, i );

496  // compute level for TFI of box inputs

507  }

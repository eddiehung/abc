204  abstractCells[1].m_pad = 0;

206  // input pads
207  Abc_NtkForEachCi( pNtk, pObj, i )
208  Abc_PlaceCreateCell( pObj, 0 );

213

216  Abc_PlaceCreateCell( pObj, 1 );
217 
218  // all nets

220  {

222  continue;

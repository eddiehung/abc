171 // p->nSatVars = 0;
172  Lit = toLit( p->nSatVars );
173  if ( p->pPars->fPolarFlip )
174  Lit = lit_neg( Lit );
175  sat_solver_addclause( p->pSat, &Lit, &Lit + 1 );
176  Dch_ObjSetSatNum( p, Aig_ManConst1(p->pAigFraig), p->nSatVars++ );
177 
178  p->nRecycles++;
179  p->nCallsSince = 0;
